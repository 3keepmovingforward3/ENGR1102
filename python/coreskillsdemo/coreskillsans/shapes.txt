def triangle(height):
    n=height 
    for i in range(n): 
        print(" "*(n-1-i)+"*"*(i+1)+"*"*(i)) 
def rectangle(length, width):
    i = 0
    n=width
    m=length
    while i < n:
        print("*"*m)
        i+=1 
    
def empty_rectangle(length, width):
    i = 0
    n=width
    m=length
    for i in range(n):
        if i == 0 or i == n-1:
            print(m*'*')
        else: 
            print('*' + ' '*(m-2) +'*')
